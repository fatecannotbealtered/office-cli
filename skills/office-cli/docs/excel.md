# Excel (.xlsx) Command Reference

Permission levels: read commands = `read-only`, write/append/create/rename/copy/style/formula/validation/sort/freeze/merge/chart/cond-format/auto-filter/column-formula/from-json/fill-range/copy-range/multi-chart/data-bar/icon-set/color-scale/hyperlink/default-font = `write`, delete-sheet = `full`.

```bash
# Discover the workbook
office-cli excel sheets sales.xlsx --json                   # one entry per worksheet (rows, cols, dimensions)
office-cli excel read sales.xlsx --json                     # whole first sheet
office-cli excel read sales.xlsx --sheet "Q1" --range A1:D10 --json
office-cli excel read sales.xlsx --sheet "Q1" --with-headers --json   # rows as {header: value} objects

# Typed read — returns native types (numbers as numbers, bools as bools) with type annotations
office-cli excel read sales.xlsx --typed --json
office-cli excel read sales.xlsx --sheet "Q1" --typed --with-headers --json

# Search across all sheets
office-cli excel search sales.xlsx "John Doe" --json
office-cli excel search sales.xlsx 100 --sheet "Q1" --limit 50 --json

# Single-cell write
office-cli excel write sales.xlsx --sheet "Q1" --ref B5 --value "1234.56"
office-cli excel write sales.xlsx --sheet "Q1" --ref C5 --formula "=SUM(B2:B4)"

# Batch write (many cells at once)
office-cli excel write sales.xlsx --cells '[
  {"sheet":"Q1","ref":"A1","value":"Name"},
  {"sheet":"Q1","ref":"B1","value":"Total"},
  {"sheet":"Q1","ref":"B5","formula":"=SUM(B2:B4)"}
]'
office-cli excel write sales.xlsx --cells @batch.json        # load JSON from a file

# Append rows (auto-detects the last used row)
office-cli excel append sales.xlsx --sheet "Q1" --rows '[["Alice", 100], ["Bob", 200]]'

# Build a brand-new workbook from spec
office-cli excel create report.xlsx --spec '{
  "sheets":[
    {"name":"Summary","rows":[["Metric","Value"],["Users",1234],["Revenue",98765]]},
    {"name":"Notes","rows":[["Generated by office-cli"]]}
  ]
}'

# Read a single cell (lighter than 'excel read' for one-cell lookups)
office-cli excel cell sales.xlsx B5 --sheet "Q1" --json

# Workbook-level info (preview optional)
office-cli excel info sales.xlsx --preview 3 --json

# CSV import / export. --bom is on by default for Windows Excel compatibility.
office-cli excel to-csv sales.xlsx --sheet "Q1" --output q1.csv --bom
office-cli excel from-csv data.csv --output data.xlsx --sheet "Imported"

# Sheet lifecycle
office-cli excel rename-sheet sales.xlsx --from "Sheet1" --to "Q1"
office-cli excel copy-sheet   sales.xlsx --from "Template" --to "Q2"
office-cli excel delete-sheet sales.xlsx --sheet "Old"

# Styling (font, fill, border, alignment, number format)
office-cli excel style sales.xlsx --sheet "Q1" --range A1:D1 --bold --bg-color "4472C4" --text-color "FFFFFF"
office-cli excel style sales.xlsx --sheet "Q1" --range B2:B100 --number-format "#,##0.00" --align right

# Batch styling — apply multiple styles in one command (single open/save for efficiency)
office-cli excel style report.xlsx --sheet "Q1" --spec '[
  {"range":"A1:J1","style":{"bold":true,"bgColor":"4472C4","textColor":"FFFFFF","align":"center"}},
  {"range":"A2:J50","style":{"numberFormat":"#,##0.00"}},
  {"range":"A51:J51","style":{"bold":true,"bgColor":"D9E2F3"}}
]' --json

# Formula generation — auto-detect data ranges and generate formulas
office-cli excel formula sales.xlsx --sheet "Q1" --col B --func SUM --json
office-cli excel formula sales.xlsx --sheet "Q1" --col B --func AVERAGE --json
office-cli excel formula sales.xlsx --sheet "Q1" --col B --func COUNTIF --criteria ">90" --json
office-cli excel formula sales.xlsx --sheet "Q1" --col B --func SUMIF --criteria "Alice" --json
# Explicit range and output cell
office-cli excel formula sales.xlsx --sheet "Q1" --col B --func SUM --data-range B2:B50 --output-cell B51 --json
# Batch: generate multiple formulas at once
office-cli excel formula sales.xlsx --sheet "Q1" --spec '[
  {"column":"B","func":"SUM"},
  {"column":"C","func":"AVERAGE"},
  {"column":"D","func":"COUNTA"}
]' --json
# Apply formulas directly to the workbook
office-cli excel formula sales.xlsx --sheet "Q1" --col B --func SUM --apply --json

# Copy workbook — deep-copy (all sheets, data, styles, charts, images)
office-cli excel copy template.xlsx --output report.xlsx --json
office-cli excel copy template.xlsx --output report.xlsx --force --json

# Data validation — dropdown lists, number/date ranges, custom formulas
office-cli excel validation form.xlsx --sheet "Input" --range D2:D100 --type list --list "Option A,Option B,Option C" --json
office-cli excel validation form.xlsx --sheet "Input" --range B2:B100 --type whole --min 0 --max 100 --json
office-cli excel validation form.xlsx --sheet "Input" --range C2:C100 --type date --min "2026-01-01" --max "2026-12-31" --json
office-cli excel validation form.xlsx --sheet "Input" --range E2:E100 --type custom --formula "=LEN(E2)>0" --json
office-cli excel validation form.xlsx --sheet "Input" --range B2:B100 --type whole --min 0 --max 100 --error-msg "Must be 0-100" --error-title "Invalid" --json

# Row / column operations
office-cli excel insert-rows sales.xlsx --sheet "Q1" --after 1 --count 5
office-cli excel insert-cols sales.xlsx --sheet "Q1" --after 2 --count 1
office-cli excel delete-rows sales.xlsx --sheet "Q1" --from 10 --count 3
office-cli excel delete-cols sales.xlsx --sheet "Q1" --from 4 --count 1
office-cli excel set-col-width sales.xlsx --sheet "Q1" --col 1 --width 30
office-cli excel set-row-height sales.xlsx --sheet "Q1" --row 1 --height 24

# Sorting
office-cli excel sort sales.xlsx --sheet "Q1" --range A1:D100 --by-col 2 --ascending

# Freeze panes (freeze top row)
office-cli excel freeze sales.xlsx --sheet "Q1" --cell A2

# Merge / unmerge cells
office-cli excel merge   sales.xlsx --sheet "Q1" --range A1:D1
office-cli excel unmerge sales.xlsx --sheet "Q1" --range A1:D1

# Charts (bar, line, pie, scatter, area, col)
office-cli excel chart sales.xlsx --sheet "Q1" --cell E2 --type bar --data-range "A1:B10" --title "Revenue"

# Conditional formatting
office-cli excel cond-format sales.xlsx --sheet "Q1" --range B2:B100 --type cell --criteria ">=" --value 1000 --bg-color "00FF00"

# Auto-filter on header row
office-cli excel auto-filter sales.xlsx --sheet "Q1" --range A1:D1

# Insert image anchored at a cell
office-cli excel add-image sales.xlsx --sheet "Q1" --cell C5 --image logo.png

# Show / hide sheets
office-cli excel hide-sheet sales.xlsx --sheet "RawData"
office-cli excel show-sheet sales.xlsx --sheet "RawData"

# ---------------------------------------------------------------------------
# Phase 2: Column formulas, JSON, range ops, multi-chart, data bar/icon set
# ---------------------------------------------------------------------------

# Column formulas — apply row-level formulas (VLOOKUP, IF, SUMIF, AVERAGEIF, CONCAT, CUSTOM)
office-cli excel column-formula report.xlsx --sheet "Data" --col D --func IF --json
# With --spec for complex formulas
office-cli excel column-formula report.xlsx --sheet "Data" --spec '{
  "column":"D","func":"IF","condition":"B{row}>150",
  "valueIfTrue":"\"High\"","valueIfFalse":"\"Low\""
}' --json
# VLOOKUP
office-cli excel column-formula report.xlsx --sheet "Data" --spec '{
  "column":"C","func":"VLOOKUP","lookupCol":"A","returnCol":"B","tableRange":"A2:B100"
}' --json
# CONCAT with separator
office-cli excel column-formula report.xlsx --sheet "Data" --spec '{
  "column":"D","func":"CONCAT","concatCols":["A","B"],"separator":" - "
}' --json
# Custom formula with {row} placeholder
office-cli excel column-formula report.xlsx --sheet "Data" --spec '{
  "column":"D","func":"CUSTOM","customFormula":"=B{row}*2"
}' --json
# Apply formulas directly (write to cells)
office-cli excel column-formula report.xlsx --sheet "Data" --col D --func IF --apply --json

# JSON conversion — export/import Excel as JSON
# Export with headers (returns [{header:value},...])
office-cli excel to-json report.xlsx --sheet "Data" --with-headers --json
# Export typed (numbers as numbers, not strings)
office-cli excel to-json report.xlsx --sheet "Data" --with-headers --typed --json
# Export as arrays (returns [[v1,v2,...],...])
office-cli excel to-json report.xlsx --sheet "Data" --json
# Import from JSON objects
office-cli excel from-json new.xlsx --sheet "Data" --data '[{"Name":"Alice","Score":95},{"Name":"Bob","Score":80}]' --json
# Import from JSON arrays
office-cli excel from-json new.xlsx --sheet "Data" --data '[["Name","Score"],["Alice",95]]' --json
# Import from file
office-cli excel from-json new.xlsx --sheet "Data" --data @data.json --json

# Range operations
# Fill a range with a single value
office-cli excel fill-range report.xlsx --sheet "Data" --range D1:D100 --value "N/A" --json
# Copy a range to another location (values + formulas)
office-cli excel copy-range report.xlsx --sheet "Data" --src A1:C10 --dst E1 --json

# Multi-series chart — chart with multiple data series
office-cli excel multi-chart report.xlsx --sheet "Data" --spec '{
  "type":"line","cell":"E2","title":"Revenue",
  "series":[
    {"name":"Sales","catRange":"A2:A13","valRange":"B2:B13"},
    {"name":"Profit","catRange":"A2:A13","valRange":"C2:C13"}
  ]
}' --json

# Data bar conditional formatting
office-cli excel data-bar report.xlsx --sheet "Data" --range B2:B100 --json
office-cli excel data-bar report.xlsx --sheet "Data" --range B2:B100 --min-color "F8696B" --max-color "63BE7B" --json

# Icon set conditional formatting
office-cli excel icon-set report.xlsx --sheet "Data" --range B2:B100 --json
office-cli excel icon-set report.xlsx --sheet "Data" --range B2:B100 --style "3TrafficLights1" --json
office-cli excel icon-set report.xlsx --sheet "Data" --range B2:B100 --style "5Arrows" --reverse --json
# Valid styles: 3Arrows, 3ArrowsGray, 3Flags, 3Signs, 3Symbols, 3Symbols2,
#               3TrafficLights1, 3TrafficLights2, 3Stars, 3Triangles,
#               4Arrows, 4ArrowsGray, 4Rating, 4RedToBlack, 4TrafficLights,
#               5Arrows, 5ArrowsGray, 5Boxes, 5Quarters, 5Rating

# Color scale (heatmap) — 2-color or 3-color gradient
office-cli excel color-scale report.xlsx --sheet "Data" --range B2:B100 --json
office-cli excel color-scale report.xlsx --sheet "Data" --range B2:B100 --min-color "F8696B" --max-color "63BE7B" --json
# 3-color scale (with mid-point)
office-cli excel color-scale report.xlsx --sheet "Data" --range B2:B100 --mid-color "FFEB84" --json

# Hyperlinks
# Set external hyperlink
office-cli excel hyperlink report.xlsx --sheet "Data" --cell A2 --link "https://example.com" --json
# Set with display text and tooltip
office-cli excel hyperlink report.xlsx --sheet "Data" --cell A2 --link "https://example.com" --display "Click here" --tooltip "Open website" --json
# Set internal link (jump to another sheet/cell)
office-cli excel hyperlink report.xlsx --sheet "Data" --cell A3 --link "Summary!A1" --type Location --json
# Read hyperlink from a cell
office-cli excel hyperlink report.xlsx --sheet "Data" --cell A2 --get --json

# Extended style flags (in addition to existing --bold/--bg-color/etc.)
office-cli excel style report.xlsx --sheet "Data" --range A1:A1 --strike --json
office-cli excel style report.xlsx --sheet "Data" --range A1:A1 --text-rotation 45 --json
office-cli excel style report.xlsx --sheet "Data" --range A1:A1 --indent 2 --json
office-cli excel style report.xlsx --sheet "Data" --range A1:A1 --shrink-to-fit --json

# Default font — get or set the workbook's default font (Calibri by default)
office-cli excel default-font report.xlsx --json                    # get current default font
office-cli excel default-font report.xlsx --set "Arial" --json      # set default font to Arial
office-cli excel default-font report.xlsx --set "Times New Roman" --json

# Cell style — read the font/style properties of a cell
office-cli excel cell-style report.xlsx A1 --sheet "Data" --json    # read style of cell A1
office-cli excel cell-style report.xlsx B5 --json                   # first sheet, read style of B5
```
